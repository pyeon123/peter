    const words = [
      { en: "apple",    ko: "사과",   roman: "sa-gwa" },
      { en: "school",   ko: "학교",   roman: "hak-gyo" },

      { en: "book",     ko: "책",     roman: "chaek" },
      { en: "computer", ko: "컴퓨터", roman: "keom-pyu-teo" }
    ];

    let currentWord = {};
    let buffer = []; // Hangul buffer

    // ===== 자모 + 영어 발음 =====
    const consonants = {
      "ㄱ": "g", "ㄲ": "kk", "ㄴ": "n", "ㄷ": "d", "ㄸ": "tt", "ㄹ": "r/l",
      "ㅁ": "m", "ㅂ": "b", "ㅃ": "pp", "ㅅ": "s", "ㅆ": "ss", "ㅇ": "ng",
      "ㅈ": "j", "ㅉ": "jj", "ㅊ": "ch", "ㅋ": "k", "ㅌ": "t", "ㅍ": "p", "ㅎ": "h"
    };
    const vowels = {
      "ㅏ":"a","ㅐ":"ae","ㅑ":"ya","ㅒ":"yae","ㅓ":"eo","ㅔ":"e","ㅕ":"yeo",
      "ㅖ":"ye","ㅗ":"o","ㅛ":"yo","ㅜ":"u","ㅠ":"yu","ㅡ":"eu","ㅣ":"i"
    };

    // ===== 음성 합성 세팅 =====
    let audioEnabled = false;
    let voices = [];
    function loadVoices() { voices = speechSynthesis.getVoices(); }
    if ('speechSynthesis' in window) {
      speechSynthesis.onvoiceschanged = loadVoices;
      loadVoices();
    }

    function enableAudio() {
      audioEnabled = true;
      document.getElementById('enableAudioBtn').disabled = true;
      document.getElementById('enableAudioBtn').textContent = "✅ Sound Enabled";
      safeSpeak("안녕하세요");
      setTimeout(() => speechSynthesis.resume?.(), 0);
    }

    function getKoreanVoice() {
      return voices.find(v => v.lang && v.lang.toLowerCase().startsWith('ko')) || null;
    }

    function safeSpeak(text, opts = {}) {
      if (!audioEnabled) return;
      if (!('speechSynthesis' in window)) return;
      try {
        speechSynthesis.cancel();
        const u = new SpeechSynthesisUtterance(text);
        u.lang = "ko-KR";
        const v = getKoreanVoice();
        if (v) u.voice = v;
        u.rate = opts.rate ?? 0.95;
        u.pitch = opts.pitch ?? 1.0;
        u.volume = opts.volume ?? 1.0;
        speechSynthesis.speak(u);
      } catch(e) {}
    }

    // ===== 게임 로직 =====
    function nextWord() {
      currentWord = words[Math.floor(Math.random() * words.length)];
      document.getElementById("word").innerText = currentWord.ko;
      document.getElementById("translation").innerText = "(" + currentWord.en + ")";
      document.getElementById("romanization").innerText = currentWord.roman;
      document.getElementById("input").value = "";
      document.getElementById("message").innerText = "";
      buffer = [];

      setTimeout(() => safeSpeak(currentWord.ko, {rate:0.9}), 100);
    }

    function checkAnswer() {
      const userAnswer = document.getElementById("input").value.trim();
      if (userAnswer === currentWord.ko) {
        document.getElementById("message").innerText = "✅ Correct!";
        safeSpeak(currentWord.ko, {rate:0.95});
        setTimeout(nextWord, 2000);
      } else {
        document.getElementById("message").innerText =
          "❌ Wrong! The correct answer is: " + currentWord.ko;
        safeSpeak("다시 시도해 보세요.");
      }
    }

    function press(key) {
      buffer.push(key);
      updateInput();
      safeSpeak(key, {rate:1.0});
    }

    function backspace() { buffer.pop(); updateInput(); }
    function space() { buffer.push(" "); updateInput(); }
    function enter() { checkAnswer(); }

    function updateInput() {
      const combined = Hangul.assemble(buffer);
      document.getElementById("input").value = combined;
    }

    function speakWord() {
      if (!currentWord.ko) return;
      safeSpeak(currentWord.ko, {rate:0.95});
    }

    // ====== 키보드 생성 ======
    function renderKeyboard() {
      const consonantDiv = document.getElementById("consonants");
      const vowelDiv = document.getElementById("vowels");
      for (const [ko, rom] of Object.entries(consonants)) {
        const btn = document.createElement("button");
        btn.innerHTML = `<span>${ko}</span><span class="roman">${rom}</span>`;
        btn.onclick = () => press(ko);
        consonantDiv.appendChild(btn);
      }
      for (const [ko, rom] of Object.entries(vowels)) {
        const btn = document.createElement("button");
        btn.innerHTML = `<span>${ko}</span><span class="roman">${rom}</span>`;
        btn.onclick = () => press(ko);
        vowelDiv.appendChild(btn);
      }
    }

    window.addEventListener('load', () => {
      renderKeyboard();
      nextWord();
